data Point:
  | point(x :: Number, y :: Number) with:
    method distance(self, other :: Point) -> Number:
      num-sqrt(num-sqr(self.x - other.x) + num-sqr(self.y - other.y))
    end
sharing:
  method is-origin(self) -> Boolean:
    (self.x == 0) and (self.y == 0)
  end
end
