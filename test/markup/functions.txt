fun fact(n):
  if n == 0:
    1
  else:
    n * fact(n - 1)
  end
where:
  fact(0) is 1
  fact(1) is 1
  fact(5) is 120
end

fun map<A, B>(f :: (A -> B), lst :: List<A>) -> List<B>:
  doc: "Map a function over a list"
  cases(List) lst:
    | empty => empty
    | link(fst, rst) => link(f(fst), map(f, rst))
  end
end
